Replacing rows with missing values (NA) with the average of the time intverval

impute <- function(x, fun) {
  missing <- is.na(x)
  replace(x, missing, fun(x[!missing]))
}
replaceNA <- ddply(data2, ~ interval, transform, steps = impute(steps, mean))
